QUESTION: 1
ANSWER:
// Write a program to print MySirG 5 times on the screen
#include<stdio.h>
int main(){
    int i=1;
    while (i<=5)
    {
        printf("MySirG\n");
        i++;
    }
    return 0;
}

QUESTION: 3
ANSWER:
// Write a program to print the first 10 natural numbers in reverse order
#include<stdio.h>
int main(){
    int i=10;
    while (i>=1)
    {
        printf("%d\n",i);
        i--;
    }
    return 0;
}
